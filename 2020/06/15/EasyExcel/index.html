二、读取

1、创建实体类用来读取数据

@Data
public class ReadData {
    //设置字段
    @ExcelProperty(value = "一级标题",index = 0)
    private String firstName;
    @ExcelProperty(value = "二级标题",index = 1)
    private String secondName;
}

2、实例操作

@Override
    public boolean readFile(MultipartFile file,EduSubjectService eduSubjectService) throws IOException {
            InputStream inputStream = file.getInputStream();
            //读取文件（使用到监听器） 参数说明：文件流,数据实体类,监听器)
            EasyExcel.read(inputStream, ReadData.class,new ExcelListener(eduSubjectService)).sheet().doRead();
            return true;
    }

3、配置监听器

继承— AnalysisEventListener<>类

**
 * 实现 ExcelListener
 */
public class ExcelListener extends AnalysisEventListener<ReadData> {
    private EduSubjectService service;
    /**
     *  获取读取excel内容
     *       :为一行一行读取
     *  readData.getFirstName()//获取第一个字段
     *  readData.getSecondName()//获取第二个字段
     */
    @Override
    public void invoke(ReadData readData, AnalysisContext analysisContext) {
        System.out.println("--->"+readData);
        EduSubject firstField = exisFirstField(readData.getFirstName());
        System.out.println(firstField);
        String id = firstField.getId();
        if(firstField==null){//如果数据库没有不重复 添加
            firstField = new EduSubject();
            firstField.setTitle(readData.getFirstName());
            firstField.setParentId("0");
            firstField.setSort(1);
            service.save(firstField);
            System.out.println("添加");
        }
        //同一转换为小写
        String secondName = readData.getSecondName().toLowerCase();
        EduSubject secondField = exisSecondField(secondName,id);
        System.out.println(secondField);
        if(secondField == null ){//如果数据库没有不重复 添加
            secondField = new EduSubject();
            secondField.setTitle(secondName);
            secondField.setParentId(firstField.getId());
            secondField.setSort(1);
            service.save(secondField);
        }
    }
    //读取完成后执行
    @Override
    public void doAfterAllAnalysed(AnalysisContext analysisContext) {
        System.out.println("添加成功");
    }
    //判断数据库中是否有一级字段
    public EduSubject exisFirstField(String name){
        QueryWrapper<EduSubject> wrapper  = new QueryWrapper<>();
        wrapper.eq("title",name);
        wrapper.eq("parent_id","0");
        EduSubject one = service.getOne(wrapper);
        return one;
    }
    //判断是否有二级字段
    public EduSubject exisSecondField(String name,String id){
        QueryWrapper<EduSubject> wrapper  = new QueryWrapper<>();
        wrapper.eq("title",name);
        wrapper.eq("parent_id",id);
        EduSubject one = service.getOne(wrapper);
        return one;
    }

    public ExcelListener(EduSubjectService service){
        this.service = service;
    }
}
